10. Display last_name and length of last-name whose names start with 'J' or 'H' or 'T'. Sort data in alphabetical order of names

SELECT
	FIRST_NAME,
	LENGTH(last_name)
FROM
	EMPLOYEES
WHERE
	FIRST_NAME LIKE 'J%'
	OR FIRST_NAME LIKE 'H%'
	OR FIRST_NAME LIKE 'T%'
ORDER BY
	FIRST_NAME;
